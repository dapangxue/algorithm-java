        StringBuilder stringBuilder = new StringBuilder();
        for (int i = 0, length = a.length(); i < length; i++) {
            if (Character.	isLetterOrDigit(a.charAt(i))) {
                // stringBuilder.append('1');
                if (b.charAt(i) == '1') {
                    count++;
                }
            } else {
                if (b.charAt(i) == '0') {
                    count++;
                }

            }
        }
        // System.out.println(stringBuilder.toString());
